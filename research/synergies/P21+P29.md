## Mathematical Integration

### Gumbel-Softmax Selection (from P21)
```
p_i = exp((f_i + g_i)/τ) / Σ_j exp((f_j + g_j)/τ)
where:
  f_i = fitness score
  g_i = Gumbel noise
  τ = temperature parameter
```

### Coevolution Diversity (from P29)
```
D = Σ_i Σ_j d(strategy_i, strategy_j) / (N*(N-1)/2)
```

## Stochastic Coevolution Algorithm

```python
class StochasticCoevolution:
    def __init__(self, solver_pop_size, generator_pop_size, initial_temp=1.0):
        self.solver_population = [Solver() for _ in range(solver_pop_size)]
        self.generator_population = [Generator() for _ in range(generator_pop_size)]
        self.temperature = initial_temp
        self.elo_system = DualPopulationELO()

    def gumbel_softmax_select(self, fitness_scores, temp=None):
        """Stochastic selection from P21 applied to coevolution."""
        if temp is None:
            temp = self.temperature

        # Add Gumbel noise
        gumbel_noise = -np.log(-np.log(np.random.uniform(0, 1, len(fitness_scores))))
        perturbed = (fitness_scores + gumbel_noise) / temp

        # Softmax
        exp_scores = np.exp(perturbed - np.max(perturbed))
        probs = exp_scores / np.sum(exp_scores)

        # Sample
        selected_idx = np.random.choice(len(fitness_scores), p=probs)
        return selected_idx, probs

    def evolve_generation(self):
        # Evaluate fitness (solver vs generator)
        solver_fitness = []
        generator_fitness = []

        for solver in self.solver_population:
            for generator in self.generator_population:
                outcome = self.compete(solver, generator)
                self.elo_system.update(solver, generator, outcome)

        # Extract fitness scores
        solver_fitness = [s.elo for s in self.solver_population]
        generator_fitness = [g.elo for g in self.generator_population]

        # Stochastic selection (P21)
        parent_indices = []
        for _ in range(len(self.solver_population)):
            idx, probs = self.gumbel_softmax_select(solver_fitness)
            parent_indices.append(idx)

        # Create offspring
        new_solvers = [self.mutate(self.solver_population[i]) for i in parent_indices]

        # Same for generators
        gen_parent_indices = [self.gumbel_softmax_select(generator_fitness)[0]
                             for _ in range(len(self.generator_population))]
        new_generators = [self.mutate(self.generator_population[i])
                         for i in gen_parent_indices]

        # Replace population
        self.solver_population = new_solvers
        self.generator_population = new_generators

        # Anneal temperature
        self.temperature *= 0.995

        return solver_fitness, generator_fitness
```

## Arms Race Detection

```python
def detect_arms_race(solver_history, generator_history):
    """Detect healthy competition via negative correlation."""
    solver_diff = np.diff(solver_history)
    generator_diff = np.diff(generator_history)

    correlation = np.corrcoef(solver_diff, generator_diff)[0, 1]

    # Negative correlation = healthy arms race
    # (when solver improves, generator must adapt and vice versa)
    is_arms_race = correlation < -0.3

    return {
        "correlation": correlation,
        "is_arms_race": is_arms_race,
        "health_score": abs(correlation)  # Higher = more competitive
    }
```

## Temperature Annealing Schedule

```python
class AdaptiveTemperatureSchedule:
    def __init__(self, initial_temp=1.0, min_temp=0.1):
        self.initial_temp = initial_temp
        self.min_temp = min_temp
        self.temperature = initial_temp

    def update_temperature(self, generation, diversity_metric, arms_race_health):
        """Adapt temperature based on population health."""
        # High diversity needed: increase temperature
        if diversity_metric < 0.5:
            self.temperature = min(self.temperature * 1.1, self.initial_temp)

        # Healthy arms race: moderate temperature
        elif arms_race_health > 0.7:
            self.temperature = max(self.temperature * 0.99, self.min_temp)

        # Low diversity: increase exploration
        else:
            self.temperature = min(self.temperature * 1.05, self.initial_temp)

        return self.temperature
```
